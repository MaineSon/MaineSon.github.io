champs <- champs %>%
    mutate(Team = ifelse(Team == "Warriorrs","Warriors",Team)) %>%
    mutate(Team = ifelse(Team == "'Heat'","Heat",Team)) %>%
    mutate(Win = as.factor(Win)) %>%
    mutate(Home = as.factor(Home))

runners <- runners %>%
    mutate(Win = as.factor(Win)) %>%
    mutate(Home = as.factor(Home))
